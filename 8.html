🏗️ Structures & Unions

Defining Structures, Array of Structures, Nested Structures, Unions, Structure vs Union

📖 Learn the Theory

What is a Structure?

A structure is a user-defined data type that groups different data types together.

Structure Syntax:

struct structure_name {
    datatype member1;
    datatype member2;
    ...
};

Declaring and Using Structures:

// Define structure
struct Student {
    int rollNo;
    char name[50];
    float marks;
};

// Create variable
struct Student s1;

// Access members using dot operator
s1.rollNo = 101;
strcpy(s1.name, "John");
s1.marks = 85.5;

Array of Structures:

struct Student class[30];  // Array of 30 students

Nested Structures (Structure within Structure):

struct Date {
    int day, month, year;
};

struct Employee {
    int id;
    char name[50];
    struct Date joiningDate;  // Nested structure
};

What is a Union?

Union is similar to structure but members share the same memory location. Only one member can hold value at a time.

Structure vs Union:

  • Structure: Each member has its own memory (sum of all members)
  • Union: All members share same memory (size of largest member)
  • Union is memory efficient but can only store one value at a time

Examples:

#include <stdio.h>
#include <string.h>

// Structure
struct Student {
    int roll;
    char name[50];
    float marks;
};

// Nested structures
struct Address {
    char city[50];
    int pincode;
};

struct Employee {
    int id;
    char name[50];
    struct Address addr;  // Nested
};

// Union
union Data {
    int i;
    float f;
    char str[20];
};

int main() {
    // Basic structure
    struct Student s1 = {101, "Alice", 85.5};
    printf("Student: %d, %s, %.2f\n", s1.roll, s1.name, s1.marks);
    
    // Array of structures
    struct Student class[3] = {
        {1, "Bob", 90},
        {2, "Carol", 88},
        {3, "David", 92}
    };
    
    for(int i = 0; i < 3; i++) {
        printf("%d: %s - %.2f\n", class[i].roll, class[i].name, class[i].marks);
    }
    
    // Nested structure
    struct Employee e1 = {1001, "John", {"Mumbai", 400001}};
    printf("Employee: %s from %s\n", e1.name, e1.addr.city);
    
    // Union demonstration
    union Data u;
    printf("Size of union: %lu bytes\n", sizeof(u));
    
    u.i = 10;
    printf("Integer: %d\n", u.i);
    
    u.f = 3.14;
    printf("Float: %.2f (int is now garbage: %d)\n", u.f, u.i);
    
    return 0;
}

Key Points to Remember:

  • Use dot (.) operator for structure variables
  • Use arrow (->) operator for structure pointers
  • Union size = size of largest member
  • Only one union member can be active at a time

✍️ Coding Practice (3 Questions)

Question 1

Create a structure for Book (title, author, price). Store 5 books and display the most expensive one.

#include <stdio.h>
#include <string.h>

struct Book {
    char title[50];
    char author[50];
    float price;
};

int main() {
    struct Book books[5] = {
        {"C Programming", "K&R", 450},
        {"Python Basics", "Smith", 350},
        {"Java Complete", "Jones", 550},
        {"Data Structures", "Lee", 480},
        {"Algorithms", "Cormen", 650}
    };
    
    int maxIndex = 0;
    for(int i = 1; i < 5; i++) {
        if(books[i].price > books[maxIndex].price)
            maxIndex = i;
    }
    
    printf("Most expensive book:\n");
    printf("Title: %s\nAuthor: %s\nPrice: %.2f",
           books[maxIndex].title, books[maxIndex].author, books[maxIndex].price);
    return 0;
}

Question 2

Write a program to add two complex numbers using structures.

#include <stdio.h>

struct Complex {
    float real;
    float imag;
};

struct Complex add(struct Complex c1, struct Complex c2) {
    struct Complex result;
    result.real = c1.real + c2.real;
    result.imag = c1.imag + c2.imag;
    return result;
}

int main() {
    struct Complex a = {3.5, 2.5};
    struct Complex b = {1.5, 4.5};
    struct Complex sum = add(a, b);
    
    printf("Sum: %.1f + %.1fi", sum.real, sum.imag);
    return 0;
}

Question 3

Demonstrate difference between structure and union (show memory sizes and behavior).

#include <stdio.h>

struct StructData {
    int i;
    float f;
    char c;
};

union UnionData {
    int i;
    float f;
    char c;
};

int main() {
    struct StructData s;
    union UnionData u;
    
    printf("Structure size: %lu bytes\n", sizeof(s));
    printf("Union size: %lu bytes\n\n", sizeof(u));
    
    // Structure - all members independent
    s.i = 10;
    s.f = 3.14;
    s.c = 'A';
    printf("Structure: i=%d, f=%.2f, c=%c\n", s.i, s.f, s.c);
    
    // Union - only last assigned is valid
    u.i = 10;
    printf("Union after i=10: i=%d\n", u.i);
    
    u.f = 3.14;
    printf("Union after f=3.14: f=%.2f, i=%d (garbage)\n", u.f, u.i);
    
    u.c = 'A';
    printf("Union after c='A': c=%c, f=%.2f (garbage)\n", u.c, u.f);
    
    return 0;
}
